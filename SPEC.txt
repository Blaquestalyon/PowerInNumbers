# app/models.py
from sqlalchemy import Column, Integer, String, Text, JSON, Enum, ForeignKey, DateTime, Float
from sqlalchemy.orm import declarative_base, relationship
import enum
from datetime import datetime

Base = declarative_base()

class RunStatus(str, enum.Enum):
    QUEUED = "queued"
    RUNNING = "running"
    COMPLETE = "complete"
    ERROR = "error"

class SectionStatus(str, enum.Enum):
    QUEUED = "queued"
    READY = "ready"
    RUNNING = "running"
    COMPLETE = "complete"
    ERROR = "error"

class GenerationRun(Base):
    __tablename__ = "generation_runs"

    id = Column(Integer, primary_key=True)
    stage = Column(Integer, nullable=False)  # 1 or 3
    status = Column(Enum(RunStatus), default=RunStatus.QUEUED)
    intake_payload = Column(JSON, nullable=False)
    core_analysis_a1a = Column(JSON)
    core_analysis_a1b = Column(JSON)
    error_log = Column(JSON)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    sections = relationship("GenerationSection", back_populates="run")

class GenerationSection(Base):
    __tablename__ = "generation_sections"

    id = Column(Integer, primary_key=True)
    run_id = Column(Integer, ForeignKey("generation_runs.id"), nullable=False)
    code = Column(String, nullable=False)  # "K-1", "G-7", etc.
    status = Column(Enum(SectionStatus), default=SectionStatus.QUEUED)
    llm_model = Column(String)
    prompt_version = Column(String)
    rag_context_ids = Column(JSON)
    html_fragment = Column(Text)
    qa_score = Column(Float)
    error = Column(Text)

    run = relationship("GenerationRun", back_populates="sections")

# app/main.py
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from .db import SessionLocal
from .models import GenerationRun, GenerationSection, RunStatus, SectionStatus
from .pipeline import enqueue_core_analysis

app = FastAPI()

class IntakeRequest(BaseModel):
    stage: int  # 1 or 3
    i1_profile: dict
    i2_context: dict
    i3_objectives: dict
    i4_constraints: dict
    i5_meta: dict

@app.post("/api/runs")
def create_run(req: IntakeRequest):
    db = SessionLocal()
    run = GenerationRun(
        stage=req.stage,
        intake_payload=req.dict(),
        status=RunStatus.QUEUED,
    )
    db.add(run)
    db.flush()

    # Pre-create section rows
    codes = (["K-1","K-2","K-3","K-4","K-5","K-6"] if req.stage == 1 else
             [f"G-{i}" for i in range(1, 16)])
    for code in codes:
        db.add(GenerationSection(run_id=run.id, code=code))

    db.commit()
    enqueue_core_analysis(run.id)  # Celery or background task
    return {"run_id": run.id}

@app.get("/api/runs/{run_id}")
def get_run(run_id: int):
    db = SessionLocal()
    run = db.query(GenerationRun).get(run_id)
    if not run:
        raise HTTPException(status_code=404, detail="Not found")

    sections = []
    for s in run.sections:
        sections.append({
            "code": s.code,
            "status": s.status.value,
            "html_fragment": s.html_fragment if s.status == SectionStatus.COMPLETE else None,
        })

    return {
        "id": run.id,
        "stage": run.stage,
        "status": run.status.value,
        "sections": sections,
    }

// src/pages/Report.tsx
import { useEffect, useState } from "react";

type SectionStatus = "queued" | "running" | "complete" | "error";

interface Section {
  code: string;
  status: SectionStatus;
  html_fragment: string | null;
}

export default function ReportPage() {
  const [sections, setSections] = useState<Section[]>([]);
  const [status, setStatus] = useState<string>("queued");
  const runId = new URLSearchParams(window.location.search).get("runId");

  useEffect(() => {
    if (!runId) return;
    const interval = setInterval(async () => {
      const res = await fetch(`/api/runs/${runId}`);
      const data = await res.json();
      setStatus(data.status);
      setSections(data.sections);
      if (data.status === "complete" || data.status === "error") {
        clearInterval(interval);
      }
    }, 1500);
    return () => clearInterval(interval);
  }, [runId]);

  return (
    <div className="min-h-screen bg-slate-950 text-slate-50">
      {/* Hero / progress */}
      <header className="max-w-5xl mx-auto py-8">
        <h1 className="text-3xl font-semibold">AI Mirror Report</h1>
        <p className="mt-2 text-slate-300">
          Status: <span className="font-mono">{status}</span>
        </p>
      </header>

      {/* Sections */}
      <main className="max-w-5xl mx-auto pb-16 space-y-8">
        {sections.map((s) => (
          <section
            key={s.code}
            id={`output-${s.code}`}
            className="border border-slate-800 rounded-xl p-6 bg-slate-900/60"
          >
            <div className="flex items-center justify-between mb-4">
              <h2 className="text-xl font-semibold">{s.code}</h2>
              <span className="text-xs uppercase tracking-wide text-slate-400">
                {s.status}
              </span>
            </div>
            {s.html_fragment ? (
              <div
                className="prose prose-invert max-w-none"
                dangerouslySetInnerHTML={{ __html: s.html_fragment }}
              />
            ) : (
              <p className="text-slate-500 text-sm">
                Generating this section…
              </p>
            )}
          </section>
        ))}
      </main>
    </div>
  );
}
